* 判断是否全选商品 ，（sca: 如果商品全选中了，商家也会跟着选中）
```
    /**
     * 判断是否全选商品 ，（sca: 如果商品全选中了，商家也会跟着选中）
     */
    private fun isCheckAllGoods(orgid: Long) {
        for (datum in data) {
            if (datum is CarEntity) {
//                找到商家 id
                if (orgid == datum.orgid) {
                    for (good in datum.subItems) {
                        if (!isDEL) {
                            if (!good.isChecked) {
                                return
                            }
                        } else {
                            if (!good.isDelChecked) {
                                return
                            }
                        }
                    }
                    if (!isDEL) {
                        datum.isChecked = true
                    } else {
                        datum.isDelChecked = true
                    }

                }
            }
        }
    }
```
